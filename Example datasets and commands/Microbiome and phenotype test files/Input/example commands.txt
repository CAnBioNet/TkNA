	python3 ./reconstruction/intake_data.py --data-dir ./testing/lactobacillus/input --out-file ./testing/lactobacillus/output/all_data_and_metadata.cdf

	python analysis/assess_network.py --file ./testing/lactobacillus/output/network_output/correlations_bw_signif_measurables.csv --source partner1 --target partner2

		python ./analysis/infomap_assignment.py --pickle ./testing/lactobacillus/output/network_output/network.pickle

	python ./random_networks/compute_network_stats.py --networks-file ./testing/lactobacillus/output/network_output/all_random_networks.pickle --bibc-groups node_types --bibc-calc-type rbc --stats-file ./testing/lactobacillus/output/network_output/random_network_analysis.pickle --node-map ./testing/lactobacillus/input/type_map_all_unique.csv --node-groups microbe pheno
