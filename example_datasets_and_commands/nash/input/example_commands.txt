	python ./analysis/infomap_assignment.py --network ./example_datasets_and_commands/nash/output/network_output/network_output_comp.csv --network-format csv --map ./example_datasets_and_commands/nash/input/type_map.csv --out-dir ./example_datasets_and_commands/nash/output/network_output/

	python ./analysis/louvain_partition.py --network ./example_datasets_and_commands/nash/output/network_output/network_output_comp.csv --network-format csv --map ./example_datasets_and_commands/nash/input/type_map.csv --out-dir ./example_datasets_and_commands/nash/output/network_output/

	python ./analysis/find_all_shortest_paths_bw_subnets.py --network ./example_datasets_and_commands/nash/output/network_output/network_output_comp.csv --network-format csv --map ./example_datasets_and_commands/nash/input/type_map.csv --node-groups lipid_metab RNAseq --out-dir ./example_datasets_and_commands/nash/output/network_output/ (**time consuming**)
